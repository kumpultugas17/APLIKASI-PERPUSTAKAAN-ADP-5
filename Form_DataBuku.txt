Private Sub UserForm_Initialize()
Dim DATA_BUKU As Long
Dim iRow As Long
iRow = Sheet2.Range("A" & Rows.Count).End(xlUp).Row
DATA_BUKU = Application.WorksheetFunction.CountA(Sheet2.Range("A4:A100000"))
If DATA_BUKU = 0 Then
    tabelData.RowSource = ""
Else
    tabelData.RowSource = "BUKU!A4:H" & iRow
End If
End Sub
